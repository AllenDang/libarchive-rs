1//! Safe Rust bindings for libarchive
2//!
3//! This crate provides idiomatic Rust bindings to libarchive, supporting reading and writing
4//! various archive formats (tar, zip, 7z, etc.) with multiple compression formats.
5//!
6//! # Examples
7//!
8//! ## Reading an archive
9//!
10//! ```no_run
11//! use libarchive2::ReadArchive;
12//!
13//! // Reading from a file (lifetime is 'static)
14//! let mut archive = ReadArchive::open("archive.tar.gz")?;
15//!
16//! while let Some(entry) = archive.next_entry()? {
17//!     println!("File: {}", entry.pathname().unwrap_or_default());
18//!     // Read entry data...
19//! }
20//! # Ok::<(), Box<dyn std::error::Error>>(())
21//! ```
22//!
23//! ## Writing an archive
24//!
25//! ```no_run
26//! use libarchive2::{WriteArchive, ArchiveFormat, CompressionFormat};
27//!
28//! let mut archive = WriteArchive::new()
29//!     .format(ArchiveFormat::Tar)
30//!     .compression(CompressionFormat::Gzip)
31//!     .open_file("output.tar.gz")?;
32//!
33//! archive.add_file("file.txt", b"Hello, world!")?;
34//! # Ok::<(), Box<dyn std::error::Error>>(())
35//! ```

66/// Returns the version string of the underlying libarchive library
67pub fn version() -> String {
68    // SAFETY: archive_version_string returns a static string that is always valid
69    unsafe {
70        let ptr = libarchive2_sys::archive_version_string();
71        std::ffi::CStr::from_ptr(ptr).to_string_lossy().into_owned()
72    }
73}
74
75/// Returns the version number of the underlying libarchive library
76pub fn version_number() -> i32 {
77    // SAFETY: Simple integer return, no pointers involved
78    unsafe { libarchive2_sys::archive_version_number() }
79}
